3   import java.net.URL;
4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.Scanner;
7   
8   public class PlayerReader implements Reader {
9   
10      private Scanner scanner;
11  
12      public PlayerReader(String pageUrl) {
13          try {
14              URL url = new URL(pageUrl);
15              scanner = new Scanner(url.openStream());
16          } catch (Exception e) {
17              e.printStackTrace();
18          }
19      }
20  
21      @Override
22      public List<Player> getPlayers() {
23          ArrayList<Player> players = new ArrayList<Player>();
24          while (scanner.hasNextLine()) {
25              String[] parts = scanner.nextLine().split(";");
26              if (parts.length > 3) {
27                  players.add(new Player(parts[0].trim(), parts[1], extractInt(parts[3]), extractInt(parts[4])));
28              }
29          }
30          return players;
31      }
32  
33      private int extractInt(String str) {
34          return Integer.parseInt(str.trim());
35      }
36  }
